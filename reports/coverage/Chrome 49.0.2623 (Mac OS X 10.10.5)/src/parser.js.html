"use strict";
 
import markdown from 'marked';
import config from './config';
import katex from 'parse-katex';
 
const defaultHighlight = config.defaultHighlight;
 
function highlight(code, callback){
    var re = /:::(\S+)\n([\s\S]+)/.exec(code);
    var lang, content;
    if(re){
        lang = re[1];
        content = re[2];
    }
    else{
        lang = defaultHighlight;
        content = code;
    }
    return require('highlight.js').highlight(lang, content).value;
}
 
const options = {
    renderer: new markdown.Renderer(),
    gfm: true,
    tables: true,
    breaks: false,
    pedantic: false,
    sanitize: true,
    smartLists: true,
    smartypants: false,
    highlight: highlight
};
 
 
export default function parse(page){
    var html = markdown.parse(
        page,
        options
    );
    return katex.renderLaTeX(html);
}
